#### 方式一：

shadowsock+v2ray

```sh
cat > /etc/shadowsocks-libev/config.json <<EOF
{
    "server":"0.0.0.0",
    "server_port":9000,
    "password":"meiyoumima",
    "timeout":300,
    "method":"chacha20-ietf-poly1305",
    "fast_open":true,
    "nameserver":"1.1.1.1",
    "mode":"tcp_only",
    "plugin":"v2ray-plugin",
    "plugin_opts":"server"
}
EOF
```

```
docker run -d -p 9000:9000  --name ss-libev --restart=always --privileged=true -v /etc/shadowsocks-libev:/etc/shadowsocks-libev teddysun/shadowsocks-libev
```

#### 方式二：

v2ray

```
#新建文件夹
mkdir -p /software/v2ray/logs
#新建配置文件
vi /software/v2ray/config.json

#编辑内容
{
  "log" : {
    "access": "/var/log/v2ray/access.log",
    "error": "/var/log/v2ray/error.log",
    "loglevel": "warning"
  },
  "inbounds": [{
    "port": 8010,
    "protocol": "vmess",
    "settings": {
      "clients": [
        {
          "id": "11111-22222-33333-44444-55555",
          "level": 1,
          "alterId": 64
        }
      ]
    }
  }],
  "outbounds": [{
    "protocol": "freedom",
    "settings": {}
  }]
}

##运行容器
docker run -d \
  --restart=always \
  --name=v2ray \
  -p 8010:8010 \
  -v /software/v2ray/config.json:/etc/v2ray/config.json \
  -v /software/v2ray/logs:/var/log/v2ray \
  v2ray/official


 ##v2ray客户端配置：

protocol选择vmess
ip填服务器ip
port填上面port
username填inbounds.settings.clients.id配置的值



注意事项：
    密码必须为uuid
```
